Storage: PostgreSQL primary (messages indexed by (session_id, created_at)). Redis for hot session state with TTL refresh.

Context Assembler

Write / Select / Compress / Isolate — prepares every LLM call.

class ContextAssembler:
    async def assemble(self, session_id, user_msg) -> AssembledContext:
        # 1. WRITE — persist user message
        await self.recorder.record_message(session_id, user_msg)

        # 2. SELECT — gather context
        recent = await self.memory.short_term.get_recent(session_id, k=20)
        history_vec = await self.memory.long_term.search(user_msg, top_k=5)
        rag_chunks = await self.retriever.query(user_msg, top_k=3)

        # 3. COMPRESS — fit within token budget
        budget = self._calc_token_budget(model_max=128_000)
        if recent.tokens > budget.recent:
            recent = await self._summarize_older_turns(recent)

        # 4. ASSEMBLE — ordered segments
        return AssembledContext(
            system=self.system_prompt,
            memory=history_vec,
            summary=recent.summary,
            recent=recent.verbatim_turns,
            rag=rag_chunks,
            tool_defs=self.tool_registry.definitions,
            user_message=user_msg,
            total_tokens=sum(s.tokens for s _)
        )

Policy Engine — Deterministic Action Router

A state-machine that maps workflow state to the next deterministic action. When the state is unambiguous, the LLM is bypassed entirely. When the engine returns None, the LLM decides. Pattern adapted from src/services/board_orchestrator/policy_engine.py. 

def evaluate_next_action(
    has_requirements: bool,
    tools_selected: bool,
    examples_checked: bool,
    has_board_draft: bool,
    has_knowledge: bool,
    is_edit_mode: bool,
    has_changes_requested: bool,
    actions_this_turn: list[dict],
    requirements_need_refinement: bool = False,
) -> dict | None:
    """
    Returns a deterministic action decision or None (let LLM decide).
    """
    done_actions = {a['action'] for a in actions_this_turn}

    # --- Refinement flow (board exists, user asked for changes) ---
    if has_changes_requested and has_board_draft:
        if 'select_tools' not in done_actions:
            return _action('select_tools', {}, 'Re-selecting tools')
        if 'select_tools' in done_actions and 'build_board' not in done_actions:
            return _action('build_board', {}, 'Applying changes to board')
        if 'build_board' in done_actions and not has_knowledge:
            return _action('build_knowledge', {}, 'Rebuilding knowledge')
        if has_knowledge and 'complete' not in done_actions:
            return _action('complete', {
                'message': 'Changes applied. Review the preview or tell me what else to change.'
            }, 'Refinement complete')

    # --- First-turn creation pipeline ---
    if not has_board_draft:
        if not has_requirements:
            return _action('gather_requirements', {}, 'No requirements yet')
        if has_requirements and not tools_selected:
            return _action('select_tools', {}, 'Requirements ready — selecting tools')
        if has_requirements and tools_selected:
            return _action('build_board', {}, 'Tools selected — building board')

    # --- Post-build pipeline ---
    if has_board_draft and not has_changes_requested:
        if not has_knowledge:
            return _action('build_knowledge', {}, 'Generating knowledge section')
        # complete: return None so LLM writes the summary

    return None  # Ambiguous — LLM decides


def _action(action, action_input, thought) -> dict:
    return {
        'action': action,
        'action_input': action_input,
        'thought': thought,
        'message_to_user': '',
        '_deterministic': True,
    }
ActionTriggered WhenEffect
gather_requirementsNo requirements extracted yetExtract structured requirements from user message
select_toolsRequirements ready, no tools selectedMatch requirements to available tools
build_boardRequirements + tools ready, no draftGenerate the board/composition
build_knowledgeBoard draft exists, no knowledge sectionPopulate knowledge/references
completeAll prior steps doneFinalize and present to user
Why deterministic routing? The LLM is expensive, slow, and inconsistent for trivial state transitions. This pattern runs in ~1μs and guarantees the pipeline advances predictably. The LLM is only consulted for ambiguous states (e.g., what to say in the completion message).

CompletenessEvaluator (deterministic, no LLM)

class CompletenessEvaluator:
    def evaluate(self, trace_data, board, config) -> CompletenessResult:
        # All 4 checks run in parallel via ThreadPoolExecutor
        scores = [
            self.check_all_tasks_executed(trace_data, board),
            self.check_all_outputs_populated(trace_data, board),
            self.check_run_completed(trace_data),
            self.check_swimlanes_completed(trace_data, board),
        ]
        overall_passed = all(s.passed for s in scores)
        overall_score = sum(s.score for s in scores) / len(scores)

        return CompletenessResult(
            overall_passed=overall_passed,
            overall_score=overall_score,
            scores=scores,
            execution_time_ms=execution_time
        )

CorrectnessEvaluator (schema + optional LLM judge + errors + ground truth)

class CorrectnessEvaluator:
    def evaluate(self, trace_data, board, config) -> CorrectnessResult:
        scores = []

        # 1. Schema validation (fast, deterministic)
        scores.append(self.validate_schemas(trace_data, board))

        # 2. LLM judge (optional, parallel batches)
        if config.get('run_llm_judge', True):
            llm_evals, llm_score = self.evaluate_with_llm_judge(...)
            if llm_score: scores.append(llm_score)

        # 3. Error detection (fast, deterministic)
        errors, error_score = self.detect_errors(trace_data)
        scores.append(error_score)

        # 4. Ground truth comparison (optional)
        if config.get('expected_outputs'):
            gt_score = self.compare_ground_truth(trace_data, config['expected_outputs'])
            scores.append(gt_score)

        overall_score = sum(s.score for s in scores) / len(scores)
        overall_passed = all(s.passed for s in scores)
        return CorrectnessResult(overall_passed, overall_score, scores, ...)

Ground Truth Comparison

def compare_ground_truth(self, trace_data, expected_outputs) -> EvaluationScore:
    """Compare final outputs against expected ground truth."""
    final_outputs = trace_data.get('output', {})
    compared_fields = 0
    matched_fields = 0

    for key, expected in expected_outputs.items():
        compared_fields += 1
        actual = final_outputs.get(key)

        if isinstance(expected, dict):
            if 'contains' in expected:       # Substring match
                if expected['contains'] in str(actual):
                    matched_fields += 1
            elif 'regex' in expected:         # Regex match
                if re.search(expected['regex'], str(actual)):
                    matched_fields += 1
        else:
            if actual == expected:                # Exact match
                matched_fields += 1

    score = matched_fields / compared_fields if compared_fields else 1.0
    return EvaluationScore(
        criterion="ground_truth_match",
        passed=matched_fields == compared_fields,
        score=score,
        details={"matched": matched_fields, "total": compared_fields}
    )

Tool Registry

class ToolRegistry:
    tools: dict[str, ToolDefinition]

    async def execute(self, name, params, idempotency_key) -> ToolResult:
        cached = await self.cache.get(idempotency_key)
        if cached:
            return cached

        try:
            result = await asyncio.wait_for(
                self._invoke(name, params), timeout=30.0
            )
        except TimeoutError:
            self.circuit_breaker.record_failure(name)
            result = ToolResult(status="error", error="timeout")

        await self.cache.set(idempotency_key, result, ttl=3600)
        return result

-- Evaluation runs (one per board/run execution)
CREATE TABLE board_run_evaluations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    run_id UUID NOT NULL REFERENCES kanban_board_runs(id) ON DELETE CASCADE,
    user_id INTEGER NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    trace_id VARCHAR(255) NOT NULL,
    board_id UUID REFERENCES kanban_boards(id) ON DELETE SET NULL,
    board_name VARCHAR(255),
    status VARCHAR(50),
    average_score DECIMAL(5, 4),
    latency_ms DECIMAL(12, 3),
    metadata JSONB,
    langfuse_url TEXT,
    fetched_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(run_id)
);

-- Individual eval scores (multiple per evaluation)
CREATE TABLE evaluation_scores (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    evaluation_id UUID NOT NULL REFERENCES board_run_evaluations(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL,
    value DECIMAL(5, 4) NOT NULL,
    comment TEXT,
    source VARCHAR(50) DEFAULT 'EVAL',
    score_created_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- RL trace evaluations (categorical pass/fail)
CREATE TABLE trace_evaluations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    run_id UUID NOT NULL,
    board_id UUID NOT NULL,
    structure_verdict VARCHAR(20),
    execution_verdict VARCHAR(20),
    completeness_verdict VARCHAR(20),
    classification VARCHAR(30),   -- GOOD_COMPLETE | GOOD_PARTIAL | BAD_STRUCTURAL

    langfuse_trace_id VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

⚙️ Workers

WorkerTriggerFunction
eval_worker.py SQS message Consumes eval job messages after board run completes. Runs Completeness + Correctness + Goal Achievement evals. Saves to DB + annotates Langfuse trace.
session_ttl.py Cron (5 min) Finds sessions with last_active_at < NOW() - TTL, sets status to expired. Evicts stale Redis keys.
memory_compaction.py Cron (daily) For sessions exceeding N messages, compresses old turns into a summary (via LLM) and stores in long-term memory.

Worker Pattern

from abc import ABC, abstractmethod

class BaseWorker(ABC):
    interval_seconds: int = 300

    async def run_forever(self):
        while True:
            try:
                await self.tick()
            except Exception as e:
                logger.error(f"tick failed: {e}")
            await asyncio.sleep(self.interval_seconds)

    @abstractmethod
    async def tick(self): ...

docker compose up -d

# 2. Run migrations
docker compose exec api alembic upgrade head

# 3. Create a session
curl -X POST http://localhost:8000/chat/session \
  -H "Authorization: Bearer sk-test-key" \
  -H "Content-Type: application/json" \
  -d '{"user_id": "user_123", "metadata": {}}'

# {"session_id": "abc-123", "status": "active"}

# 4. Send a message (streaming)
curl -N -X POST http://localhost:8000/chat/message \
  -H "Authorization: Bearer sk-test-key" \
  -H "Content-Type: application/json" \

  -H "Authorization: Bearer sk-test-key" \
  -H "Content-Type: application/json" \
  -d '{"expected_outputs": {"status": "completed"}}'
